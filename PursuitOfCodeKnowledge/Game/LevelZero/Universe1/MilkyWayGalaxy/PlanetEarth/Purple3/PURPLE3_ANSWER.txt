def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        # Флаг для оптимизации: если не происходит обменов, массив отсортирован
        swapped = False
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]  # Меняем местами
                swapped = True
        if not swapped:  # Если не было обменов, выходим из цикла
            break
    return arr

def main():
    robots = ["T1111", "T2222", "T3333", "T4444", "T5555", "T6666", "T7777", "T0000"]
    print(robots)
    bubble_sort(robots)
    print(robots)
if __name__ == "__main__":
    main()
