ls

ls -a

ls ../Blue/BLUE3_ANSWER.txt

	y = 3

	# Сложение
	sum_result = x + y

	# Вычитание
	difference = x - y

	# Умножение
	product = x * y

	# Деление
	quotient = x / y

	# Целочисленное деление
	integer_division = x // y

	# Остаток от деления
	remainder = x % y

	# Возведение в степень
	power = x ** y

	print("Сумма:", sum_result)
	print(f"Разность: {difference}")
	print("Произведение:", product)
	print(f"Частное: {quotient}")
	print("Целочисленное деление:", integer_division)
	print(f"Остаток от деления: {remainder}")
	print("Возведение в степень:", power)

	a = 5
	b = 10

	# Сравнение
	print("a меньше b:", a < b)
	print(f"a больше b: {a > b}")
	print("a равно b:", a == b)
	print(f"a не равно b: {a != b}")
	print("a меньше или равно b:", a <= b)
	print(f"a больше или равно b: {a >= b}")

	is_a_true = True
	is_b_false = False

	print("is_a_true и is_b_false:", is_a_true and is_b_false)  # Вернет False
	print(f"is_a_true или is_b_false: {is_a_true or is_b_false})  # Вернет True
	print("не is_a_true:", not is_a_true)      # Вернет False
	print("не is_b_false:", not is_b_false)      # Вернет True
