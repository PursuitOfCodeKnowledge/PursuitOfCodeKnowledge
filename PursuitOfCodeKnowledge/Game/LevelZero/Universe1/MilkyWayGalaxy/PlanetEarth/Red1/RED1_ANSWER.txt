def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_index = i  # Предполагаем, что текущий элемент — минимальный
        for j in range(i + 1, n):
            if arr[j] < arr[min_index]:
                min_index = j  # Находим индекс минимального элемента
        arr[i], arr[min_index] = arr[min_index], arr[i]  # Меняем местами
    return arr


def main():
    robots = ["T1111", "T2222", "T3333", "T4444", "T5555", "T6666", "T7777", "T0000"]
    print(robots)
    selection_sort(robots)
    print(robots)
if __name__ == "__main__":
    main()
