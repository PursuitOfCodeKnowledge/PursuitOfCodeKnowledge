	return f"Привет, {name}!"

def add_numbers(*args):
	return sum(args)

print("Сумма:", add_numbers(1, 2, 3, 4, 5))  # Вывод: Сумма: 15

def print_info(**kwargs):
	for key, value in kwargs.items():
		print(f"{key}: {value}")

print_info(name="Искуственный интеллект", age=18)  # Вывод: name: Искусственный интеллект, age: 18

	def adder(y):
		return x + y
	return adder

print(add_five(10))  # Вывод: 15

""" Лямбда-функции """
double = lambda x: x * 2
print(double(10))  # Вывод: 20
