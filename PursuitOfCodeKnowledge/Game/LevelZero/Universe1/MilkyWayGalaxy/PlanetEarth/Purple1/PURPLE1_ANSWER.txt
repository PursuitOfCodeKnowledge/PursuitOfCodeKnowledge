def binary_search(arr, target):
    left, right = 0, len(arr) - 1

    while left <= right:
        mid = left + (right - left) // 2  # Находим середину
        if arr[mid] == target:
            return mid  # Если элемент найден, возвращаем индекс
        elif arr[mid] < target:
            left = mid + 1  # Если элемент больше середины, ищем справа
        else:
            right = mid - 1  # Если элемент меньше середины, ищем слева
    return -1  # Если элемент не найден, возвращаем -1

def main():
    robots = ["T1111", "T2222", "T3333", "T4444", "T5555", "T6666", "T7777", "T0000"]
    target = "T7777"
    robots.sort()
    print(binary_search(robots, target))
if __name__ == "__main__":
    main()
